1) class Solution:
    def makeGood(self, s: str) -> str:
        stack = []

        for c in s:
            if stack and abs(ord(stack[-1]) - ord(c)) == 32:
                stack.pop()
            else:
                stack.append(c)

        return "".join(stack)

        for i, node in enumerate(lists):
            if node:
                heapq.heappush(heap, (node.val, i, node))
